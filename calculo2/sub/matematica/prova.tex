	\item Calcule a integral $\ds\int \frac{5x+1}{(2x+1)(x-1)}\ dx$.

	\vspace{0.5cm}

		\[\int_{-2}^1 x^{-4}\ dx = \left. \frac{x^{-3}}{-3} \right|_{-2}^1 = -\frac{3}{8}\]
		\begin{enumerate}

\end{enumerate}
